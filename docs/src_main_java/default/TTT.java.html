import java.util.Scanner;

public class TTT {
	Scanner scanner = new Scanner(System.in);
	private Board board = new Board();
	private Rules rules = new Rules();
	private Player player_1 = new Player("Player 1", 'X');
	private Player player_2 = new Player("Player 2", 'O');
	protected Player current_player = player_1;
	
	public TTT() {}
	
	public void start() {
		showTitle();
		boolean run = true;
		
		while(run) {
			board.printBoard();
			
			while(!getUserRowCol());
			board.setSymbolAt(current_player.getCol(), current_player.getCol(), current_player.getSymbol());
			
			if (rules.playerWon(board)) {
				System.out.println("\n" + current_player.getName() + " won!\n");
				current_player.incrScore();
				run = rematch();
				
			}else if(board.isFull()) {
				System.out.print("Board is full, game is a draw!");
				run = rematch();
				
			}else {
				changePlayer();
			}
		}
	}
	
	private boolean rematch() {
		scanner.nextLine();
		System.out.print("Play again (y/n): ");
		char input = scanner.next().charAt(0);
		if(input == 'y') {
			board.initBoard();
			return true;
		}else {
			return false;
		}
	}
	
	private boolean getUserRowCol() {
		System.out.println("\n" + current_player.getName() + ", place your marker '" + current_player.getSymbol() + "' in an empty spot!");
		System.out.print("Row: ");
		current_player.setRow(scanner.nextInt()-1);
		System.out.print("Col: ");
		current_player.setCol(scanner.nextInt()-1);
		return board.validateRowCol(current_player.getRow(), current_player.getCol());
	}
	
	private void showTitle() {
		System.out.println("Tic Tac Toe \n");
		System.out.println(player_1.getName() + " score: " + player_1.getScore());
		System.out.println(player_2.getName() + " score: " + player_2.getScore());
		System.out.println("");
	}
	
	public void changePlayer() {
		if(current_player == player_1) {
			this.current_player = player_2;
		}else {
			this.current_player = player_1;
		}
	}
}
